1    import java.io.*;
2    public class SelectionSort {
3        void sort(int arr[]) {
4            int n= arr.length;
5            for (int i= 0; i < n-1; i++) {
6                //min element in unsorted arr
7                int min_indx = i;
8                for (int j= i + 1; j < n; j++)
9                    if (arr[j] < arr[min_indx])
10                       min_indx= j;
11               //swapping of the min element with 1st element
12               int temp= arr[min_indx];
13               arr[min_indx]= arr[i];
14               arr[i] = temp;
15           }
16       }
17       //print
18       void printArray(int arr[]) {
19           int n= arr.length;
20           for (int i = 0; i < n; ++i)
21               System.out.print(arr[i]+ " ");
22           System.out.println();
23       }

25       public static void main(String args[]) {
